I focus on scalable environment construction and agent self-evolution, driving continuous improvement in reasoning and action capabilities through the synergy of data and environment synthesis and evaluation.

\section{\faFileText\ Representative Papers}
{\small \textit{* denotes equal contribution, $^\dagger$ denotes corresponding author.}}  

\noindent \href{https://arxiv.org/abs/2607.05943}{\textbf{SearchEyes: Towards Frontier Multimodal Deep Search Intelligence via Search World Simulation}} \hfill [1] \\
\textit{arXiv preprint arXiv:2607.05943.} \\
\underline{Zhengbo Jiao}, Yiming Cheng, Yilei Jiang, Kaituo Feng, Rui Huang, Tianyi Jiang, Juanxi Tian, Qunzhong Wang, Tailai Chen, Qianshan Wei, Chuan Xiao, Shanyu Rong, Yangfu Li, Yanhan Zhou, Yifan Zhang, Xiangyu Yue\textsuperscript{†}. \\
\noindent \textbf{TL;DR:} We present \textbf{SearchEyes}, which uses a typed knowledge graph as a simulated search world unifying data synthesis, environment simulation, and RL training. We propose \textbf{PKC} and \textbf{HaPO} for step-level credit assignment. \textbf{SearchEyes-27B} achieves \textbf{SOTA} among open-source multimodal search agents, improving over the strongest baseline by \textbf{6.2 pts} on average across 6 benchmarks.

\noindent \href{https://arxiv.org/abs/2602.03279}{\textbf{Agentic Proposing: Enhancing Large Language Model Reasoning via Compositional Skill Synthesis}} \hfill [2]\\
\textit{\textcolor{red}{Accepted by ICML 2026.}} arXiv preprint arXiv:2602.03279. \\
\underline{Zhengbo Jiao}, Shaobo Wang, Zifan Zhang, Xuan Ren, Wei Wang, Bing Zhao, Hu Wei\textsuperscript{†}, Linfeng Zhang\textsuperscript{†}. \\
\noindent \textbf{TL;DR:} We propose \textbf{Agentic Proposing}, redefining problem construction as \textbf{compositional skill synthesis}. Equipped with \textbf{atomic, composable skills}, our agent orchestrates modules via \textit{tool use} and \textit{reflection} to generate \textit{scalable}, \textit{high-difficulty} problems. A \textbf{30B model} on 11k trajectories achieves \textbf{91.6\% on AIME25}, rivaling frontier models.

\noindent \href{https://arxiv.org/abs/2509.24726}{\textbf{Socratic-Zero: Bootstrapping Reasoning via Data-Free Agent Co-evolution}} \hfill [3] \\
\textit{arXiv preprint arXiv:2509.24726.} \textcolor{red}{Reported by Machine Heart}. \\
\underline{Zhengbo Jiao}\textsuperscript{*}, Shaobo Wang\textsuperscript{*}, Zifan Zhang, Yilang Peng, Xu Ze, Boyu Yang, Wei Wang, Hu Wei\textsuperscript{†}, Linfeng Zhang\textsuperscript{†}. \\
\noindent \textbf{TL;DR:} We propose \textbf{Socratic-Zero}, a dynamic data synthesis framework with a \textbf{Teacher--Solver} closed loop targeting the Solver's weaknesses. A lightweight \textbf{Generator} via \textbf{value-weighted distillation} focuses on \textit{50\% success-rate} problems for automatic curriculum learning, delivering \textbf{+20.2 points} gain and outperforming commercial LLMs.

\noindent \href{https://arxiv.org/abs/2601.20379}{\textbf{Policy of Thoughts: Scaling LLM Reasoning via Test-time Policy Evolution}} \hfill [4]\\
\textit{\textcolor{red}{Accepted by EMNLP 2026.}} arXiv preprint arXiv:2601.20379. \\
\underline{Zhengbo Jiao}, Hongyu Xian, Qinglong Wang, Yunpu Ma, Zhebo Wang, Zifan Zhang, Dezhang Kong, Meng Han\textsuperscript{†}. \\
\noindent \textbf{TL;DR:} Current inference scaling suffers from \textit{frozen policy} assumptions. We introduce \textbf{Policy of Thoughts (PoT)}, recasting reasoning as \textbf{within-instance online optimization}: \textbf{MCTS} explores conjectures while \textbf{GRPO} updates a \textbf{transient LoRA adapter} from feedback. A \textbf{4B model} achieves \textbf{49.5\% on LiveCodeBench}, surpassing models with \textbf{50$\times$ parameters}.

\noindent \href{https://arxiv.org/abs/2602.03414}{\textbf{Socratic-Geo: Synthetic Data Generation and Geometric Reasoning via Multi-Agent Interaction}} \hfill [5] \\
\textit{\textcolor{red}{Accepted by CVPR 2026.}} arXiv preprint arXiv:2602.03414. \\
\underline{Zhengbo Jiao}\textsuperscript{*}, Shaobo Wang\textsuperscript{*}, Zifan Zhang, Wei Wang, Bing Zhao, Hu Wei\textsuperscript{†}, Linfeng Zhang\textsuperscript{†}. \\
\noindent \textbf{TL;DR:} We propose \textbf{Socratic-Geo}, replacing passive collection with \textbf{programmatic rendering driven by an Agentic Teacher}. Our agent modifies \textbf{executable Python scripts} to synthesize geometrically precise problems with \textit{Reflect-RePI} self-verification. This enables scalable image-text pair creation, achieving \textbf{+4.13 pts} reasoning gain and \textbf{42.4\% on GenExam-Math}.
